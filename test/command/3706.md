Results marker can be hidden in block attributes (#3706)

```
pandoc -f org -t native
#+BEGIN_SRC R :exports results :colnames yes
   data.frame(Id = 1:3, Desc = rep("La",3))
#+END_SRC

#+CAPTION: Lalelu.
#+LABEL: tab
#+RESULTS:
| Id | Desc |
|----+------|
|  1 | La   |
|  2 | La   |
|  3 | La   |
^D
[Div ("tab",[],[])
 [Table ("",[],[]) (Caption Nothing
  [Para [Str "Lalelu."]]) [(AlignDefault,Nothing),(AlignDefault,Nothing)] 0
  [Row ("",[],[])
   [Cell ("",[],[]) Nothing 1 1
    [Plain [Str "Id"]]
   ,Cell ("",[],[]) Nothing 1 1
    [Plain [Str "Desc"]]]]
  [Row ("",[],[])
   [Cell ("",[],[]) Nothing 1 1
    [Plain [Str "1"]]
   ,Cell ("",[],[]) Nothing 1 1
    [Plain [Str "La"]]]
  ,Row ("",[],[])
   [Cell ("",[],[]) Nothing 1 1
    [Plain [Str "2"]]
   ,Cell ("",[],[]) Nothing 1 1
    [Plain [Str "La"]]]
  ,Row ("",[],[])
   [Cell ("",[],[]) Nothing 1 1
    [Plain [Str "3"]]
   ,Cell ("",[],[]) Nothing 1 1
    [Plain [Str "La"]]]]
  []]]
```
